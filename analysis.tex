\chapter{Real Analysis}

\begin{df}[Cantor Set]
  Define a sequence of sets $A_n \subseteq [0, 1]$ by:
  \begin{align*}
    A_0 & = [0,1] \\
    A_{n+1} & = A_n \setminus \bigcup_{k=0}^{3^{n-1} - 1}
    \left( \frac{3k+1}{3^n},     \frac{3k+2}{3^n} \right)
  \end{align*}
  The \emph{Cantor set} is $\bigcap_{n=0}^\infty A_n$.
\end{df}
